# 快速开始

// 引入样式
import '@pingtoug/vxui/lib/css/vxui.css'

import VXUI from '@pingtoug/vxui'
Vue.use(VXUI)
// 或者
import { XButton } from '@pingtoug/vxui'
Vue.use(XButton)
